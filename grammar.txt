stmt    ->   ident = expr ;
           | if ( expr ) stmt
           | if ( expr ) stmt else stmt
           | for ( optexpr ; optexpr ; optexpr) stmt

optexpr -> epsilon | expr
expr    -> expr oper term | term
term    -> digit | ident
oper    -> + | - | * | / | =
digit   -> 0..9
ident   -> a..z
